/* src/pages/Waitlist.jsx */
import React, { useState } from 'react'

function getList(){ try{ return JSON.parse(localStorage.getItem('grapefi_waitlist')||'[]') }catch(e){return[]} }
function saveList(l){ localStorage.setItem('grapefi_waitlist', JSON.stringify(l)) }

export default function Waitlist(){
  const [name, setName] = useState('')
  const [email, setEmail] = useState('')
  const [sol, setSol] = useState('')
  const [msg, setMsg] = useState('')
  const [list, setList] = useState(getList())

  function add(e){
    e.preventDefault();
    if(!sol || sol.length < 32 || sol.length > 44){ setMsg('Please enter a valid Solana address'); return }
    const entry = {name, email, sol, added: new Date().toISOString()}
    const next = [...list, entry]
    saveList(next)
    setList(next)
    setName(''); setEmail(''); setSol(''); setMsg('Added!')
    setTimeout(()=>setMsg(''),4000)
  }

  function downloadCSV(){
    if(!list.length){ alert('No entries'); return }
    const header = ['name','email','sol','added']
    const rows = list.map(r => header.map(h => JSON.stringify(r[h]||'')).join(','))
    const csv = [header.join(','), ...rows].join('\n')
    const blob = new Blob([csv], {type:'text/csv'})
    const url = URL.createObjectURL(blob)
    const a = document.createElement('a'); a.href = url; a.download = 'grapefi_waitlist.csv'; document.body.appendChild(a); a.click(); a.remove(); URL.revokeObjectURL(url)
  }

  return (
    <div className="max-w-xl">
      <h2 className="font-pixel text-2xl mb-4">Join the Waitlist</h2>
      <form onSubmit={add} className="space-y-3">
        <input value={name} onChange={e=>setName(e.target.value)} placeholder="Name / Discord (opt)" className="w-full p-3 rounded-md bg-white/5" />
        <input value={email} onChange={e=>setEmail(e.target.value)} placeholder="Email (opt)" className="w-full p-3 rounded-md bg-white/5" />
        <input value={sol} onChange={e=>setSol(e.target.value)} placeholder="Solana address (required)" className="w-full p-3 rounded-md bg-white/5" required />
        <div className="flex gap-3">
          <button type="submit" className="bg-gradient-to-b from-grape4 to-grape3 px-4 py-2 rounded-md">Join Waitlist</button>
          <button type="button" onClick={downloadCSV} className="bg-white/5 px-4 py-2 rounded-md">Download CSV</button>
        </div>
        {msg && <div className="text-sm text-accent">{msg}</div>}
      </form>

      {list.length>0 && (
        <div className="mt-6 bg-white/3 p-3 rounded-md text-sm text-gray-200">
          <div className="mb-2">Saved entries: {list.length}</div>
          <div className="max-h-48 overflow-auto text-xs">
            {list.map((l,i)=> (
              <div key={i} className="py-1 border-b border-white/5">{l.sol} — {l.name || '—'} — {l.email || '—'}</div>
            ))}
          </div>
        </div>
      )}
    </div>
  )
}
